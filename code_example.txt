def snowflake(lengthSide,levels) {
    if(levels == 0) {
        forward(lengthSide);
        return();
    }
    lengthSide=lengthSide/3.0;
    snowflake(lengthSide,levels-1);
    left(60);
    snowflake(lengthSide,levels-1);
    right(120);
    snowflake(lengthSide,levels-1);
    left(60);
    snowflake(lengthSide,levels-1);
}
snowflake(50,5);
